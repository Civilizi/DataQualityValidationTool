interface DomainState {
  currentDomainId: string | null;
  domains: Array<{ id: string; name: string; status: string }>;
  setCurrentDomain: (id: string) => void;
  loadDomains: () => Promise<void>;
}

export const useDomainStore = create<DomainState>((set) => ({
  currentDomainId: null,
  domains: [],
  setCurrentDomain: (id) => set({ currentDomainId: id }),
  loadDomains: async () => {

    const json = await res.json();
    if (json.success) set({ domains: json.data });
  },
}));

### 统一请求封装

```typescript
// src/lib/api.ts
export async function apiRequest<T>(
  method: string,
  path: string,
  body?: unknown
): Promise<{ success: boolean; data?: T; error?: { code: string; message: string } }> {
  const options: RequestInit = { method, headers: {} };
  if (body) {
    options.headers!["Content-Type"] = "application/json";
    options.body = JSON.stringify(body);
  }
  const res = await fetch(path, options);
  return res.json();
}
```

### AI 解析流程

```
上传 Excel → POST /api/standards (multipart)
    ↓
标准状态 = "pending"
    ↓
用户点击"开始解析" → POST /api/standards/[id]/parse
    ↓
标准状态 = "parsing" → 前端每 3 秒轮询 GET /api/standards/[id]
    ↓
标准状态 = "parsed" → 自动跳转到规则预览页面
    ↓
规则预览页面：展示所有 pending 规则，低置信度优先排序
    ↓
用户逐条/批量确认 → POST /api/rules/batch-confirm
    ↓
标准状态 = "confirmed"，规则状态 = "confirmed"
```

### 版本对比组件

```typescript
interface VersionDiffProps {
  versionA: Rule[];
  versionB: Rule[];
}

// 对比两个版本的规则差异：新增、删除、修改
const computeDiff = (a: Rule[], b: Rule[]) => {
  const mapA = new Map(a.map((r) => [r.id, r]));
  const mapB = new Map(b.map((r) => [r.id, r]));

  const added = b.filter((r) => !mapA.has(r.id));
  const removed = a.filter((r) => !mapB.has(r.id));
  const modified = b.filter((r) => {
    const old = mapA.get(r.id);
    return old && JSON.stringify(old) !== JSON.stringify(r);
  });

  return { added, removed, modified };
};

// 使用 Ant Design Descriptions 展示差异
<Descriptions bordered>
  <Descriptions.Item label="新增规则">{diff.added.length} 条</Descriptions.Item>
  <Descriptions.Item label="删除规则">{diff.removed.length} 条</Descriptions.Item>
  <Descriptions.Item label="修改规则">{diff.modified.length} 条</Descriptions.Item>
</Descriptions>
```

### 技术要点

1. **AI 解析为异步**：前端轮询 `parseStatus`，间隔 3 秒
2. **规则表格排序**：低置信度（红色）优先展示
3. **置信度标签颜色**：高=绿、中=橙、低=红
4. **批量操作**：支持勾选多条规则后批量确认/驳回
5. **版本对比**：使用 diff 算法对比两个版本的规则差异，以不同颜色标注

### 分块上传组件

```typescript
import SparkMD5 from "spark-md5";

interface UploadState {
  file: File | null;
  fileHash: string;
  sessionId: string | null;
  totalChunks: number;
  uploadedChunks: number[];
  uploading: boolean;
  progress: number;
  status: "idle" | "hashing" | "init" | "uploading" | "merging" | "done" | "error";
}

const CHUNK_SIZE = 5 * 1024 * 1024; // 5MB
const CONCURRENCY = 3; // 3 并发

const ChunkUploader = ({ domainId, onSuccess }: Props) => {
  const [state, setState] = useState<UploadState>({...});

  // 1. 计算文件 MD5
  const calculateHash = async (file: File): Promise<string> => {
    return new Promise((resolve) => {
      const spark = new SparkMD5.ArrayBuffer();
      const reader = new FileReader();
      let currentChunk = 0;
      const chunks = Math.ceil(file.size / CHUNK_SIZE);

      reader.onload = (e) => {
        spark.append(e.target!.result as ArrayBuffer);
        currentChunk++;
        if (currentChunk < chunks) {
          loadNext();
        } else {
          resolve(spark.end());
        }
      };

      const loadNext = () => {
        const start = currentChunk * CHUNK_SIZE;
        const end = Math.min(start + CHUNK_SIZE, file.size);
        reader.readAsArrayBuffer(file.slice(start, end));
      };

      loadNext();
    });
  };

  // 2. 初始化上传会话
  const initUpload = async (file: File, hash: string) => {
    const res = await apiRequest("POST", "/api/assets/upload/init", {
      fileName: file.name,
      fileSize: file.size,
      fileHash: hash,
      chunkSize: CHUNK_SIZE,
      domainId,
    });
    if (res.success) {
      // 秒传命中
      if (res.data.instantUpload) {
        setState(prev => ({ ...prev, status: "done", sessionId: res.data.assetId }));
        onSuccess?.(res.data);
        return;
      }
      // 检查 localStorage 恢复
      const saved = localStorage.getItem(`upload_${hash}`);
      const restoredChunks = saved ? JSON.parse(saved) : [];

      setState(prev => ({
        ...prev,
        sessionId: res.data.sessionId,
        totalChunks: res.data.totalChunks,
        uploadedChunks: restoredChunks.length > 0 ? restoredChunks : res.data.uploadedChunks,
        status: "uploading",
      }));
    }
  };

  // 3. 并发上传分片
  const uploadChunks = async (file: File) => {
    const totalChunks = state.totalChunks;
    const uploaded = new Set(state.uploadedChunks);

    for (let i = 0; i < totalChunks; i += CONCURRENCY) {
      const batch = [];
      for (let j = 0; j < CONCURRENCY && i + j < totalChunks; j++) {
        const chunkIndex = i + j;
        if (uploaded.has(chunkIndex)) continue;

        const start = chunkIndex * CHUNK_SIZE;
        const end = Math.min(start + CHUNK_SIZE, file.size);
        const chunk = file.slice(start, end);

        batch.push(uploadSingleChunk(state.sessionId!, chunkIndex, chunk));
      }

      await Promise.all(batch);

      // 保存进度到 localStorage
      const allUploaded = Array.from({ length: totalChunks }, (_, k) => k);
      localStorage.setItem(`upload_${state.fileHash}`, JSON.stringify(allUploaded));
    }
  };

  const uploadSingleChunk = async (sessionId: string, chunkIndex: number, chunk: Blob) => {
    const formData = new FormData();
    formData.append("sessionId", sessionId);
    formData.append("chunkIndex", String(chunkIndex));
    formData.append("file", chunk);

    const res = await fetch("/api/assets/upload/chunk", { method: "POST", body: formData });
    return res.json();
  };

  // 4. 完成合并
  const completeUpload = async () => {
    setState(prev => ({ ...prev, status: "merging" }));
    const res = await apiRequest("POST", "/api/assets/upload/complete", {
      sessionId: state.sessionId,
    });
    if (res.success) {
      setState(prev => ({ ...prev, status: "done" }));
      localStorage.removeItem(`upload_${state.fileHash}`);
      onSuccess?.(res.data);
    }
  };

  // 完整上传流程
  const handleUpload = async (file: File) => {
    setState(prev => ({ ...prev, file, status: "hashing" }));
    const hash = await calculateHash(file);
    setState(prev => ({ ...prev, fileHash: hash }));
    await initUpload(file, hash);
    if (state.status !== "done") {
      await uploadChunks(file);
      await completeUpload();
    }
  };

  return (
    <Upload.Dragger
      beforeUpload={(file) => { handleUpload(file); return false; }}
      showUploadList={false}
    >
      <p className="ant-upload-drag-icon"><InboxOutlined /></p>
      <p className="ant-upload-text">点击或拖拽文件到此区域上传</p>
      <p className="ant-upload-hint">支持 .xlsx / .xls 格式，大文件自动分块上传</p>
    </Upload.Dragger>
  );
};
```

### 技术要点

1. **API Key 密码框脱敏**：使用 `Input.Password` 组件，编辑模式下部分隐藏（如 `sk-****...****`）
2. **Temperature 滑块**：范围 0-1，步长 0.1
3. **Prompt 按类型分类**：`standard_parse` / `field_match` / `issue_diagnosis` / `report_gen`
4. **Prompt 版本回退**：点击回退后调用 `POST /api/settings/prompts/[id]/rollback` 传目标版本号
5. **别名批量导入**：支持粘贴 JSON 或下载模板 Excel 后上传

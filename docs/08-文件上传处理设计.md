## 8.1 分块上传流程

大文件（最高 500MB）通过分块切片 + 断点续传机制上传，确保网络中断后可恢复，避免全量重传。

```
前端                              后端
  |                                 |
  |-- 1. 用户选择文件                |
  |                                 |
  |-- 2. SparkMD5 计算文件 MD5       |
  |     (使用 Web Worker)            |
  |                                 |
  |-- 3. POST /api/assets/upload/init
  |     { fileName, fileSize, fileHash, chunkSize, domainId }
  |<-- { sessionId, uploadedChunks: [], totalChunks }
  |                                 |
  |   4. 检查 uploadedChunks        |
  |      找出缺失的分片索引          |
  |                                 |
  |   5. 并发上传缺失分片（3 并发）   |
  |-- 6. POST /api/assets/upload/chunk
  |     multipart/form-data:
  |       sessionId, chunkIndex, file
  |<-- { success: true, uploadedChunks: [0,1,...] }
  |                                 |
  |   7. 重复步骤 5-6 直到全部上传完  |
  |                                 |
  |-- 8. POST /api/assets/upload/complete
  |     { sessionId }
  |                                 |   9. 校验分片完整性
  |                                 |   10. 按序合并分片
  |                                 |       → uploads/assets/
  |                                 |   11. 解析 Excel 元数据
  |                                 |   12. 创建 data_assets 记录
  |                                 |   13. 检查 fileHash 是否重复（秒传）
  |<-- { asset: { id, name, version, ... } }
```

### 8.1.1 流程说明

| 步骤 | 说明 |
|------|------|
| 1 | 用户在文件选择器中选择文件 |
| 2 | 前端使用 SparkMD5 库计算文件 MD5 哈希值，作为文件唯一标识 |
| 3 | 初始化上传会话：后端检查是否已有同名文件上传过（秒传），返回已上传分片列表 |
| 4 | 前端对比已上传分片，只上传缺失部分 |
| 5-6 | 并发上传缺失分片，控制并发数避免浏览器卡顿 |
| 7 | 循环直到所有分片上传完成 |
| 8 | 通知后端合并分片 |
| 9-10 | 后端校验分片完整性后按序合并 |
| 11 | 使用 SheetJS 解析合并后 Excel 的元数据（Sheet 名、行数、列名） |
| 12 | 创建 `data_assets` 数据库记录 |
| 13 | 如果 fileHash 已存在（同一文件重复上传），执行秒传逻辑 |

## 8.2 前端上传实现要点

### 8.2.1 核心上传类

```typescript
// lib/upload/chunk-uploader.ts

const CHUNK_SIZE = 5242880; // 5MB
const UPLOAD_CONCURRENCY = 3;
const CHUNK_TIMEOUT_MS = 60000;
const MAX_RETRIES = 2;

class ChunkUploader {
  private file: File;
  private sessionId: string | null = null;
  private uploadedChunks: Set<number> = new Set();
  private totalChunks: number = 0;

  constructor(file: File) {
    this.file = file;
    this.totalChunks = Math.ceil(file.size / CHUNK_SIZE);
  }

  /**
   * 1. 计算文件 MD5（使用 Web Worker 避免阻塞 UI）
   */
  async calculateMD5(file: File): Promise<string> {
    return new Promise((resolve, reject) => {
      const worker = new Worker('/workers/md5-worker.js');
      worker.postMessage({ file, chunkSize: CHUNK_SIZE });

      worker.onmessage = (e) => {
        if (e.data.progress !== undefined) {
          // 更新进度条
          updateProgress('md5', e.data.progress);
        }
        if (e.data.md5) {
          worker.terminate();
          resolve(e.data.md5);
        }
      };

      worker.onerror = (error) => {
        worker.terminate();
        reject(error);
      };
    });
  }

  /**
   * 2. 初始化上传会话
   */
  async init(fileName: string, fileSize: number, fileHash: string, domainId: string): Promise<{
    sessionId: string;
    uploadedChunks: number[];
    totalChunks: number;
  }> {
    const response = await fetch('/api/assets/upload/init', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ fileName, fileSize, fileHash, chunkSize: CHUNK_SIZE, domainId }),
    });

    if (!response.ok) {
      throw new Error(`初始化上传失败：${response.statusText}`);
    }

    const data = await response.json();
    this.sessionId = data.sessionId;
    this.uploadedChunks = new Set(data.uploadedChunks);
    this.totalChunks = data.totalChunks;

    // 保存到 localStorage 以便断点恢复
    localStorage.setItem(`upload_${fileHash}`, JSON.stringify({
      sessionId: data.sessionId,
      uploadedChunks: data.uploadedChunks,
      fileName,
      fileSize,
      fileHash,
    }));

    return data;
  }

  /**
   * 3. 上传单个分片
   */
  async uploadChunk(sessionId: string, chunkIndex: number, chunk: Blob): Promise<void> {
    const formData = new FormData();
    formData.append('sessionId', sessionId);
    formData.append('chunkIndex', String(chunkIndex));
    formData.append('file', chunk);

    const controller = new AbortController();
    const timeoutId = setTimeout(() => controller.abort(), CHUNK_TIMEOUT_MS);

    try {
      const response = await fetch('/api/assets/upload/chunk', {
        method: 'POST',
        body: formData,
        signal: controller.signal,
      });

      clearTimeout(timeoutId);

      if (!response.ok) {
        throw new Error(`分片 ${chunkIndex} 上传失败：${response.statusText}`);
      }

      const data = await response.json();
      this.uploadedChunks = new Set(data.uploadedChunks);

      // 更新 localStorage
      this.saveProgress();
    } catch (error) {
      clearTimeout(timeoutId);
      throw error;
    }
  }

  /**
   * 4. 并发上传（控制并发数）
   */
  async uploadChunks(
    sessionId: string,
    chunks: { index: number; blob: Blob }[],
    concurrency: number = UPLOAD_CONCURRENCY
  ): Promise<void> {
    let currentIndex = 0;

    const uploadNext = async (): Promise<void> => {
      if (currentIndex >= chunks.length) return;

      const chunk = chunks[currentIndex++];
      let retries = 0;

      while (retries <= MAX_RETRIES) {
        try {
          await this.uploadChunk(sessionId, chunk.index, chunk.blob);
          updateProgress('upload', (this.uploadedChunks.size / this.totalChunks) * 100);
          break;
        } catch (error) {
          retries++;
          if (retries > MAX_RETRIES) {
            throw error;
          }
          await new Promise(resolve => setTimeout(resolve, 1000 * retries));
        }
      }
    };

    // 并发执行
    const workers: Promise<void>[] = [];
    for (let i = 0; i < Math.min(concurrency, chunks.length); i++) {
      workers.push(uploadNext());
    }
    await Promise.all(workers);
  }

  /**
   * 5. 完成上传
   */
  async complete(sessionId: string): Promise<Asset> {
    const response = await fetch('/api/assets/upload/complete', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ sessionId }),
    });

    if (!response.ok) {
      throw new Error(`完成上传失败：${response.statusText}`);
    }

    const data = await response.json();

    // 清理 localStorage
    localStorage.removeItem(`upload_${data.asset.fileHash}`);

    return data.asset;
  }

  /**
   * 6. 断点恢复
   */
  async resume(fileHash: string): Promise<boolean> {
    const saved = localStorage.getItem(`upload_${fileHash}`);
    if (!saved) return false;

    const data = JSON.parse(saved);
    this.sessionId = data.sessionId;
    this.uploadedChunks = new Set(data.uploadedChunks);
    this.totalChunks = Math.ceil(data.fileSize / CHUNK_SIZE);

    return true;
  }

  /**
   * 保存进度到 localStorage
   */
  private saveProgress() {
    if (this.sessionId) {
      localStorage.setItem(`upload_${this.sessionId}`, JSON.stringify({
        sessionId: this.sessionId,
        uploadedChunks: Array.from(this.uploadedChunks),
        totalChunks: this.totalChunks,
      }));
    }
  }

  /**
   * 完整上传流程
   */
  async upload(domainId: string, onProgress: (phase: string, percent: number) => void): Promise<Asset> {
    globalThis.updateProgress = onProgress;

    // 步骤 1：计算 MD5
    const fileHash = await this.calculateMD5(this.file);

    // 步骤 2：初始化
    const initResult = await this.init(
      sanitizeFileName(this.file.name),
      this.file.size,
      fileHash,
      domainId
    );

    // 步骤 3：找出缺失分片
    const missingChunks: { index: number; blob: Blob }[] = [];
    for (let i = 0; i < this.totalChunks; i++) {
      if (!this.uploadedChunks.has(i)) {
        const start = i * CHUNK_SIZE;
        const end = Math.min(start + CHUNK_SIZE, this.file.size);
        missingChunks.push({ index: i, blob: this.file.slice(start, end) });
      }
    }

    // 步骤 4：并发上传缺失分片
    if (missingChunks.length > 0) {
      await this.uploadChunks(this.sessionId!, missingChunks);
    }

    // 步骤 5：完成上传
    return this.complete(this.sessionId!);
  }
}
```

### 8.2.2 MD5 Web Worker

```javascript
// public/workers/md5-worker.js
import SparkMD5 from 'spark-md5';

self.onmessage = async function (e) {
  const { file, chunkSize } = e.data;
  const spark = new SparkMD5.ArrayBuffer();
  let offset = 0;

  while (offset < file.size) {
    const chunk = await file.slice(offset, offset + chunkSize).arrayBuffer();
    spark.append(chunk);
    offset += chunkSize;

    // 报告进度
    const progress = Math.round((offset / file.size) * 100);
    self.postMessage({ progress });
  }

  const md5 = spark.end();
  self.postMessage({ md5 });
};
```

```typescript
// lib/upload/sanitize.ts

export function sanitizeFileName(fileName: string): string {
  return fileName
    .replace(/[<>:"/\\|?*\x00-\x1F]/g, '')  // 移除危险字符

    .substring(0, 200);                        // 限制长度
}
```

```typescript
// /api/assets/upload/init

export async function POST(req: Request) {
  const { fileName, fileSize, fileHash, chunkSize, domainId } = await req.json();

  // 检查秒传：同一 hash 是否已存在
  const existingAsset = await db.query.dataAssets.findFirst({
    where: and(
      eq(dataAssets.fileHash, fileHash),
      eq(dataAssets.domainId, domainId),
    ),
  });

  if (existingAsset) {
    return Response.json({
      quickUpload: true,
      asset: existingAsset,
      message: '文件已存在，已秒传',
    });
  }

  // 检查是否有未完成的上次上传（断点续传）
  const existingSession = await db.query.uploadSessions.findFirst({
    where: and(
      eq(uploadSessions.fileHash, fileHash),
      eq(uploadSessions.status, 'pending'),
    ),
  });

  const totalChunks = Math.ceil(fileSize / chunkSize);

  if (existingSession) {
    // 续传：返回已有会话和已上传分片
    return Response.json({
      sessionId: existingSession.sessionId,
      uploadedChunks: existingSession.uploadedChunks,
      totalChunks,
      quickUpload: false,
    });
  }

  // 新会话
  const sessionId = crypto.randomUUID();
  const session = await db.insert(uploadSessions).values({
    sessionId,
    fileName,
    fileSize,
    fileHash,
    chunkSize,
    totalChunks,
    uploadedChunks: [],
    domainId,
    status: 'pending',
    createdAt: new Date(),
    updatedAt: new Date(),
  }).returning();

  return Response.json({
    sessionId,
    uploadedChunks: [],
    totalChunks,
    quickUpload: false,
  });
}
```

### 8.3.3 接收分片

```typescript
// /api/assets/upload/chunk

import { writeFile, mkdir } from 'fs/promises';
import path from 'path';

const CHUNKS_DIR = path.join(process.cwd(), 'uploads', 'chunks');

export async function POST(req: Request) {
  const formData = await req.formData();
  const sessionId = formData.get('sessionId') as string;
  const chunkIndex = parseInt(formData.get('chunkIndex') as string, 10);
  const file = formData.get('file') as File;

  if (!sessionId || isNaN(chunkIndex) || !file) {
    return Response.json({ error: '缺少必要参数' }, { status: 400 });
  }

  // 查找会话
  const session = await db.query.uploadSessions.findFirst({
    where: eq(uploadSessions.sessionId, sessionId),
  });

  if (!session) {
    return Response.json({ error: '会话不存在' }, { status: 404 });
  }

  // 保存分片
  const chunkDir = path.join(CHUNKS_DIR, sessionId);
  await mkdir(chunkDir, { recursive: true });

  const chunkPath = path.join(chunkDir, `${chunkIndex}`);
  const buffer = Buffer.from(await file.arrayBuffer());
  await writeFile(chunkPath, buffer);

  // 更新已上传分片列表
  const uploadedChunks = [...new Set([...session.uploadedChunks, chunkIndex])];
  await db.update(uploadSessions)
    .set({ uploadedChunks, updatedAt: new Date() })
    .where(eq(uploadSessions.sessionId, sessionId));

  return Response.json({ success: true, uploadedChunks });
}
```

### 8.3.4 完成上传

```typescript
// /api/assets/upload/complete

import { readFile, readdir, unlink, rmdir } from 'fs/promises';
import { createWriteStream } from 'fs';

const ASSETS_DIR = path.join(process.cwd(), 'uploads', 'assets');

export async function POST(req: Request) {
  const { sessionId } = await req.json();

  // 1. 查找会话
  const session = await db.query.uploadSessions.findFirst({
    where: eq(uploadSessions.sessionId, sessionId),
  });

  if (!session) {
    return Response.json({ error: '会话不存在' }, { status: 404 });
  }

  // 2. 验证分片完整性
  if (session.uploadedChunks.length !== session.totalChunks) {
    return Response.json({
      error: '分片不完整',
      uploaded: session.uploadedChunks.length,
      total: session.totalChunks,
    }, { status: 400 });
  }

  // 3. 按序合并分片
  const chunkDir = path.join(CHUNKS_DIR, sessionId);
  const mergedFileName = `${session.fileHash}_${Date.now()}_${sanitizeFileName(session.fileName)}`;
  const mergedPath = path.join(ASSETS_DIR, mergedFileName);
  const writeStream = createWriteStream(mergedPath);

  for (let i = 0; i < session.totalChunks; i++) {
    const chunkPath = path.join(chunkDir, `${i}`);
    const chunkBuffer = await readFile(chunkPath);
    writeStream.write(chunkBuffer);
    await unlink(chunkPath); // 合并后删除分片
  }

  writeStream.end();
  await new Promise<void>((resolve, reject) => {
    writeStream.on('finish', resolve);
    writeStream.on('error', reject);
  });

  // 清理分片目录
  await rmdir(chunkDir);

  // 4. 校验合并后文件 MD5
  const mergedBuffer = await readFile(mergedPath);
  const actualHash = crypto.createHash('md5').update(mergedBuffer).digest('hex');
  if (actualHash !== session.fileHash) {
    // MD5 不匹配，删除文件
    await unlink(mergedPath);
    return Response.json({ error: '文件完整性校验失败' }, { status: 400 });
  }

  // 5. 解析 Excel 元数据
  const excelMeta = parseExcelMetadata(mergedBuffer);

  // 6. 检查 fileHash 是否已存在（第二次秒传检查）
  const existingAsset = await db.query.dataAssets.findFirst({
    where: and(
      eq(dataAssets.fileHash, session.fileHash),
      eq(dataAssets.domainId, session.domainId),
    ),
  });

  if (existingAsset) {
    // 已存在，删除新文件，返回已有资产
    await unlink(mergedPath);
    return Response.json({ quickUpload: true, asset: existingAsset });
  }

  // 7. 获取下一个版本号
  const existingVersions = await db.query.dataAssets.findMany({
    where: and(
      eq(dataAssets.name, session.fileName),
      eq(dataAssets.domainId, session.domainId),
    ),
  });
  const version = existingVersions.length > 0
    ? Math.max(...existingVersions.map(a => a.version)) + 1
    : 1;

  // 8. 创建 data_assets 记录
  const asset = await db.insert(dataAssets).values({
    name: session.fileName,
    filePath: mergedPath,
    fileHash: session.fileHash,
    fileSize: session.fileSize,
    sheetCount: excelMeta.sheets.length,
    sheetMetadata: excelMeta.sheets,
    version,
    domainId: session.domainId,
    status: 'active',
    createdAt: new Date(),
    updatedAt: new Date(),
  }).returning();

  // 9. 更新会话状态
  await db.update(uploadSessions)
    .set({ status: 'completed', updatedAt: new Date() })
    .where(eq(uploadSessions.sessionId, sessionId));

  return Response.json({ quickUpload: false, asset: asset[0] });
}
```

## 8.4 Excel 解析（SheetJS）

### 8.4.1 解析器

```typescript
// lib/excel/parser.ts

import * as XLSX from 'xlsx';

export interface SheetInfo {
  name: string;
  headers: string[];
  rowCount: number;
}

export interface ParsedExcel {
  sheets: SheetInfo[];
}

export function parseExcel(fileBuffer: Buffer): ParsedExcel {
  const workbook = XLSX.read(fileBuffer, { type: 'buffer' });

  return {
    sheets: workbook.SheetNames.map(name => {
      const sheet = workbook.Sheets[name];
      const data = XLSX.utils.sheet_to_json(sheet, { header: 1, defval: '' }) as any[][];

      return {
        name,
        headers: (data[0] || []).map((h: any) => String(h).trim()),
        rows: data.slice(1),
        rowCount: Math.max(0, data.length - 1),
      };
    }),
  };
}

/**
 * 解析 Excel 元数据（用于上传完成后的轻量解析）
 */
export function parseExcelMetadata(fileBuffer: Buffer): { sheets: SheetInfo[] } {
  return parseExcel(fileBuffer);
}

/**
 * 按 Sheet 名获取数据
 */
export function getSheetData(
  parsed: ParsedExcel,
  sheetName: string
): { headers: string[]; rows: any[][] } | null {
  const sheet = parsed.sheets.find(s => s.name === sheetName);
  if (!sheet) return null;
  return { headers: sheet.headers, rows: sheet.rows };
}
```

### 8.4.2 数据转记录格式

```typescript
/**
 * 将 [[header1, header2, ...], [val1, val2, ...], ...] 格式
 * 转换为 [{ header1: val1, header2: val2 }, ...] 格式
 */
export function rowsToRecords(
  headers: string[],
  rows: any[][]
): Record<string, any>[] {
  return rows.map(row => {
    const record: Record<string, any> = {};
    headers.forEach((header, idx) => {
      record[header] = row[idx] !== undefined ? row[idx] : '';
    });
    return record;
  });
}
```

## 8.5 Excel 生成（带颜色标记的导出）

校验完成后，将校验结果写入 Excel 文件，问题单元格使用颜色标记。

### 8.5.1 颜色映射

| 等级 | 背景色 | 字体色 | 含义 |
|------|--------|--------|------|
| error | `#FFC7CE` | `#9C0006` | 严重（空值、格式不符） |
| warning | `#FFEB9C` | `#9C6500` | 警告（疑似占位符） |
| info | `#BDD7EE` | `#2B579A` | 提示（业务合理性待确认） |

### 8.5.2 导出实现

```typescript
// lib/excel/generator.ts

import * as XLSX from 'xlsx';

const SEVERITY_COLORS = {
  error: { fill: '#FFC7CE', font: '#9C0006' },
  warning: { fill: '#FFEB9C', font: '#9C6500' },
  info: { fill: '#BDD7EE', font: '#2B579A' },
};

export interface ExportOptions {
  parsed: ParsedExcel;                  // 原始数据
  results: ValidationResult[];          // 校验结果
  sheetName: string;                    // 要导出的 Sheet 名
  includeComments?: boolean;            // 是否包含批注
}

export function generateMarkedExcel(options: ExportOptions): Buffer {
  const { parsed, results, sheetName, includeComments = true } = options;

  // 按 sheet 名分组校验结果
  const resultsBySheet = results.filter(r => r.sheetName === sheetName);

  // 构建问题索引：(rowIndex, fieldName) → 问题信息
  const issueMap = new Map<string, ValidationResult>();
  for (const result of resultsBySheet) {
    if (result.fieldName && result.rowIndex) {
      const key = `${result.rowIndex}-${result.fieldName}`;
      // 同一单元格多个问题时取最高等级
      const existing = issueMap.get(key);
      if (!existing || severityRank(result.severity) < severityRank(existing.severity)) {
        issueMap.set(key, result);
      }
    }
  }

  // 找到原始 sheet 数据
  const sourceSheet = parsed.sheets.find(s => s.name === sheetName);
  if (!sourceSheet) {
    throw new Error(`找不到 Sheet：${sheetName}`);
  }

  // 创建工作簿
  const workbook = XLSX.utils.book_new();
  const worksheet = XLSX.utils.aoa_to_sheet([
    sourceSheet.headers,
    ...sourceSheet.rows,
  ]);

  // 应用颜色标记
  const headerRow = 0; // SheetJS 行号从 0 开始
  for (const [key, issue] of issueMap) {
    const [rowIdxStr, fieldName] = key.split('-');
    const rowIdx = parseInt(rowIdxStr, 10) - 1; // Excel 行号转 SheetJS 索引
    const colIdx = sourceSheet.headers.indexOf(fieldName);

    if (rowIdx < 0 || colIdx < 0) continue;

    const cellRef = XLSX.utils.encode_cell({ r: rowIdx + 1, c: colIdx }); // +1 跳过表头
    const cell = worksheet[cellRef];

    if (!cell) continue;

    const colors = SEVERITY_COLORS[issue.severity];
    cell.s = {
      fill: { fgColor: { rgb: colors.fill.replace('#', '') } },
      font: { color: { rgb: colors.font.replace('#', '') }, bold: true },
    };

    // 添加批注
    if (includeComments) {
      if (!cell.c) cell.c = [];
      cell.c.push({
        a: '系统',
        t: `[${issue.severity === 'error' ? '严重' : issue.severity === 'warning' ? '警告' : '提示'}] ${issue.message}`,
      });
    }
  }

  // 添加表头样式
  for (let c = 0; c < sourceSheet.headers.length; c++) {
    const cellRef = XLSX.utils.encode_cell({ r: 0, c });
    if (worksheet[cellRef]) {
      worksheet[cellRef].s = {
        font: { bold: true },
        fill: { fgColor: { rgb: 'D9E1F2' } },
      };
    }
  }

  XLSX.utils.book_append_sheet(workbook, worksheet, sheetName);

  // 生成 Buffer
  return XLSX.write(workbook, { type: 'buffer', bookType: 'xlsx' });
}

function severityRank(severity: string): number {
  const ranks = { error: 0, warning: 1, info: 2 };
  return ranks[severity as keyof typeof ranks] ?? 3;
}
```

### 8.6.1 文件类型限制

```typescript
const ALLOWED_EXTENSIONS = ['.xlsx', '.xls', '.csv'];
const ALLOWED_MIME_TYPES = [
  'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet',
  'application/vnd.ms-excel',
  'text/csv',
];

function validateFileType(fileName: string, mimeType?: string): boolean {
  const ext = '.' + fileName.split('.').pop()?.toLowerCase();
  if (!ALLOWED_EXTENSIONS.includes(ext)) return false;
  if (mimeType && !ALLOWED_MIME_TYPES.includes(mimeType)) return false;

}
```

### 8.6.4 临时分片清理

```typescript
// 定时任务：清理超过 24 小时未完成的分片
export async function cleanupExpiredChunks(): Promise<void> {
  const expiredThreshold = Date.now() - 24 * 60 * 60 * 1000;

  // 查找超过 24 小时未完成的上传统
  const expiredSessions = await db.query.uploadSessions.findMany({
    where: and(
      eq(uploadSessions.status, 'pending'),
      lt(uploadSessions.createdAt, new Date(expiredThreshold)),
    ),
  });

  for (const session of expiredSessions) {
    // 删除分片目录
    const chunkDir = path.join(CHUNKS_DIR, session.sessionId);
    try {
      await fs.rm(chunkDir, { recursive: true, force: true });
    } catch {
      // 目录可能已被清理
    }

    // 更新会话状态
    await db.update(uploadSessions)
      .set({ status: 'expired', updatedAt: new Date() })
      .where(eq(uploadSessions.sessionId, session.sessionId));
  }
}
```

### 8.6.5 敏感数据加密标识

```typescript
// 在 data_assets 表中标记包含敏感数据的字段
interface SensitiveFieldMarker {
  assetId: string;
  fieldName: string;
  sensitiveType: 'phone' | 'id_card' | 'bank_account' | 'email' | 'address';
  encrypted: boolean;  // true = 该字段已加密存储
}

// 上传后自动检测敏感字段
export function detectSensitiveFields(headers: string[]): SensitiveFieldMarker[] {
  const patterns: Record<string, RegExp> = {
    phone: /(手机|电话|mobile|phone)/i,
    id_card: /(身份证|id.?card|证件号)/i,
    bank_account: /(银行|bank|账号|账户)/i,
    email: /(邮箱|email|mail)/i,
    address: /(地址|address|住址)/i,
  };

  const markers: SensitiveFieldMarker[] = [];
  for (const header of headers) {
    for (const [type, pattern] of Object.entries(patterns)) {
      if (pattern.test(header)) {
        markers.push({
          assetId: '', // 待填充
          fieldName: header,
          sensitiveType: type as SensitiveFieldMarker['sensitiveType'],
          encrypted: true,
        });
        break;
      }
    }
  }
  return markers;
}
```
